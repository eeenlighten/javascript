06. switch문

{
    const num = 100;

    switch(num){
        case 100 :
            document.write("실행되었습니다(num==100)");
        break;
        case 200 :
            document.write("실행되었습니다(num==200)");
        break;
        case 300 :
            document.write("실행되었습니다(num==300)");
        break;
        case 400 :
            document.write("실행되었습니다(num==400)");
        break;
        default :
            document.write("실행되었습니다.(num==값이없음");
    }
}

11. forEach()

{
    {
        const num = [100, 200, 300, 400, 500];
    
        document.write(num[0]);
        document.write(num[1]);
        document.write(num[2]);
        document.write(num[3]);
        document.write(num[4]);
    
        //for문
        for(let i=0; i<num.length; i++){
            document.write(num[i]); // 100200300400500
        }
        
    

        num.forEach(function(el){
            document.write(el); // 100200300400500
        });
    
        //forEach(element, index, array)
        num.forEach((element, index, array) => {
            document.write(element);
            document.write(index);
            document.write(array);

            // 100 0 100200300400500
            // 200 0 100200300400500
            // 300 0 100200300400500
            // 400 0 100200300400500
            // 500 0 100200300400500              
        });
}

14. break문

{
    //while문을 이용해서 0~20까지 출력
    let num = 1;
    while(num <= 20){
        document.write(num);
        if(num == 10){
            break;
        }
        num++;
    }
    
    //for문을 이용해서 0~20 출력
    for(let i = 1; i <= 20; i++){
        document.write(i);
        if(i == 10){
            break;
        }
    }
}

15. continue문

{
    let num = 0;
    while(num < 10){
        num++;
        if ( num == 5){
            continue;
        }
        document.write(num);
    }

    for(let i = 1; i<=10; i++){
        if( i == 5){
            continue;
        }
        document.write(i);
    }
}
